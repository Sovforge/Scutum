Get the API URL:
{{- if .Values.gateway.enabled }}
{{- range .Values.gateway.hostnames }}
  https://{{ . }}
{{- end }}
{{- else if eq .Values.service.api.type "LoadBalancer" }}
  export LB=$(kubectl get svc {{ include "scutum.fullname" . }} -n {{ .Release.Namespace }} \
    -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
  echo "https://$LB:{{ .Values.service.api.port }}"
{{- else if eq .Values.service.api.type "NodePort" }}
  export NODE_IP=$(kubectl get nodes -o jsonpath='{.items[0].status.addresses[0].address}')
  export NODE_PORT=$(kubectl get svc {{ include "scutum.fullname" . }} -n {{ .Release.Namespace }} \
    -o jsonpath='{.spec.ports[0].nodePort}')
  echo "https://$NODE_IP:$NODE_PORT"
{{- else }}
  kubectl port-forward -n {{ .Release.Namespace }} svc/{{ include "scutum.fullname" . }} 8080:{{ .Values.service.api.port }}
  echo "https://localhost:8080"
{{- end }}
